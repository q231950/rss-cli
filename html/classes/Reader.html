def initialize args
  setup_properties
  write_welcome_message
  parse_argv_if_available(args)

  unless @help.nil?
    show_help
  end

  unless @read.nil?
    read
  end

  if @help.nil? && @read.nil?
    show_basic_usage_hint
  end

  puts "Bye bye."
end

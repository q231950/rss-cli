def initialize args
  setup_properties
  write_welcome_message
  parse_argv_if_available(args)

  unless @help.nil?
    show_help
  end

  unless @read.nil?
    read
  end

  if @help.nil? && @read.nil?
    show_basic_usage_hint
  end

  puts "Bye bye."
end

def read
  @cli.say("Your RSS feeds:")
  @cli.choose do |menu|
    menu.prompt = "Choose a feed or quit by entering the letter 'q' or 'quit'."
    @configuration.feeds.each do |feed|
      nameSymbol = feed['name'].to_sym
      menu.choices(nameSymbol) {
        handle_name_url(feed['name'], feed['url'])
      }
    end
    menu.hidden(:quit, :q) {
    }
  end
end
